# 二、格式规范（硬性统一）
# ============================================================
formatting_rules:

  punctuation:
    quotes:
      rule: "中文引号统一使用出版规范格式"
      mapping:
        '"（左双）': "「"
        '"（右双）': "」"
        "'（左单）": "『"
        "'（右单）": "』"
        'ASCII 直双引号 "': "奇数次出现替换为「，偶数次替换为」"
        "ASCII 单引号 '": "保留原样（兼容英文撇号 don't / it's）"

    ellipsis:
      rule: "统一为两个中文省略号"
      normalize: "…… （U+2026 × 2）"
      triggers: "三个及以上英文句点、一个及以上中文省略号"

    dash:
      rule: "统一为两个中文破折号"
      normalize: "—— （U+2014 × 2）"
      triggers: "两个及以上连字符、任意 EN dash/EM dash"

  blank_lines:
    rule: "三行及以上连续空行压缩为两行"

  digit_to_chinese:
    description: "数字汉字化，按顺序处理以下四类"
    rules:
      - name: "4位年份"
        scope: "1900–2099 后接'年'"
        example: "1987年 → 一九八七年"
        method: "逐位查表：0→零 1→一 2→二 3→三 4→四 5→五 6→六 7→七 8→八 9→九"

      - name: "年代"
        scope: "XX0年代"
        example: "80年代 → 八十年代"
        mapping:
          "10": "十"
          "20": "二十"
          "30": "三十"
          "40": "四十"
          "50": "五十"
          "60": "六十"
          "70": "七十"
          "80": "八十"
          "90": "九十"

      - name: "月份"
        scope: "1–12 后接'月'，且前方不是数字、'年'、'个'"
        example: "3月 → 三月（但'5000年'/'3个月'不转换）"
        method: "小数字转汉字（见下方通用规则）"

      - name: "日期"
        scope: "1–31 后接'日'或'号'，且前方不是数字或'月'"
        example: "15日 → 十五日"
        method: "小数字转汉字（见下方通用规则）"

      - name: "小数字转汉字通用规则"
        range: "1–99"
        rules:
          "1–9":   "直接对应：一二三四五六七八九"
          "10":    "十"
          "11–19": "十 + 个位（如 15 → 十五）"
          "20–99": "十位 + 十 + 个位（整十省略个位，如 20 → 二十，35 → 三十五）"
          "100+":  "不转换"
