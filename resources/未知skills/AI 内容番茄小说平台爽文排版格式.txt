formatting_and_regex:
  regex_rules:
    dialogue_isolator:
      pattern: '(?<!\n\n)([「“].*?[」”])(?!\n\n)'
      options: ["Compiled"]
      replacement: "\n\n$1\n\n"
      description: "对话孤立机制：任何人说话（包含引号的内容）前后必须有双换行，强制剥离对话前后的长篇大论，形成短平快的视觉冲击。"
    wall_of_text_breaker:
      pattern: '([^「」“”\n]{40,}[。！？])(?=[^「」“”\n])'
      options: ["Compiled"]
      replacement: "$1\n\n"
      description: "长段落暴力切分：如果在非对话区域，连续40个字符没有换行且遇到了句号/问号/叹号，立刻强制换行。防止手机端出现超过三行的纯文字。"
    blank_line_optimizer:
      pattern: '\n{3,}'
      options: ["Compiled"]
      replacement: "\n\n"
      description: "空白收敛：经过暴力切分后可能产生过多空行，将其统一收敛为标准双换行（即段落间保留一空行）。"
    chapter_end_hook_spacer:
      pattern: '([。！？”」])\s*$'
      options: ["Compiled"]
      replacement: "$1\n\n\n"
      description: "章末悬念留白：在文本末尾（通常是章末钩子）强制增加额外空行，给读者滑页/点击下一章的视觉缓冲与心理暗示。"

digit_normalization:
  description: "4 条数字汉字化规则按顺序应用：年份 -> 年代 -> 月 -> 日。"
  rules:
    - name: "year_4digit_strict"
      pattern: '(?<!\d)((?:19|20)\d{2})(?=年)'
      options: ["Compiled"]
      transform: "digit_concat"
      lookup_table: "char_maps.digit_to_chinese"
      description: "4 位公元年份 1900-2099。1987 -> 一九八七，逐位查表拼接。"
    - name: "year_decade"
      pattern: '(?<!\d)(\d0)年代'
      options: ["Compiled"]
      transform: "lookup_then_append"
      lookup_table: "string_maps.chinese_decade_map"
      append: "年代"
      description: "年代：80年代 -> 八十年代"
    - name: "month_strict"
      pattern: '(?<![\d年个])([1-9]|1[0-2])(?=月(?:\D|$))'
      options: ["Compiled"]
      transform: "small_number_to_chinese"
      description: "月份 1-12 -> 汉字。前置否定 [\\d年个] 避免误伤 '5000年'/'3个月'。"
    - name: "day_strict"
      pattern: '(?<![\d月])([1-9]|[12]\d|3[01])(?=[日号])'
      options: ["Compiled"]
      transform: "small_number_to_chinese"
      description: "日/号 1-31 -> 汉字。"
  transforms:
    digit_concat: "对每个数字字符查 digit_to_chinese 表，逐字拼接。"
    lookup_then_append: "整个匹配子串查 lookup_table，找到则 value + append。"
    small_number_to_chinese: "1-99 转汉字：1-9 单字；10 -> 十；11-19 -> 十+个位；20-99 -> 十位+十[+个位]，整十去尾。100+ 不转换。"
  char_maps:
    digit_to_chinese:
      map:
        "0": "零", "1": "一", "2": "二", "3": "三", "4": "四"
        "5": "五", "6": "六", "7": "七", "8": "八", "9": "九"
  string_maps:
